var output = new SimpleImage(320,320);

for (var p of output.values()){
    var x = p.getX();
    var y = p.getY();

     x=x-160;
     y=160-y;

    if(((x*x+y*y-1)*(x*x+y*y-1)*(x*x+y*y-1)-x*x*y*y*y)===0){
        p.setRed(255);                                
    }
    
}
