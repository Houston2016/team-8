function setBlack(pixel){
    pixel.setRed(0);
    pixel.setGreen(0);
    pixel.setBlue(0);
}
function pixelOnEdgeDifferentThicknesses(pixel,image,horizontalWidth,verticalWidth){
    var y = pixel.getY();
    if(y<verticalWidth){
        return true;
    }
    if(y>image.getHeight()-verticalWidth){
        return true;
    }
    var x = pixel.getX();
    if(x<horizontalWidth){
        return true;
    }

    if(x>image.getWidth()-horizontalWidth){
        return true;
    }
    return false;
}

var image = new SimpleImage("eastereggs.jpg");
var HorizontalWidth = 10;
var VerticalWidth = 15;
for (var p of image.values()){
    if(pixelOnEdgeDifferentThicknesses(p,image,HorizontalWidth,VerticalWidth)){
        setBlack(p);
    }

}
