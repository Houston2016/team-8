var image = new SimpleImage("hilton.jpg" );
for(var p of image.values()){
    var x = p.getRed();
    var y = p.getGreen();
    var z = p.getBlue();
    p.setRed(255);
    if(p.getX()>50){
        p.setGreen(255);
        p.setRed(x);
        p.setBlue(z);
    }
    if(p.getX()>100){
        p.setBlue(255);
        p.setRed(x);
        p.setGreen(y);
    }
}
print(image);
